# Dockerfile.dev - Optimized for development speed
FROM python:3.11-slim

# Install system dependencies (minimal for dev)
RUN apt-get update && apt-get install -y \
    gcc \
    && rm -rf /var/lib/apt/lists/*

WORKDIR /app

# Install Python dependencies (cached layer)
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Install development tools
RUN pip install --no-cache-dir \
    flask-shell-ipython \
    ipython \
    ipdb \
    watchdog

# Create non-root user for development
RUN groupadd -r appuser && useradd -r -g appuser appuser
RUN chown -R appuser:appuser /app
USER appuser

# Source code is mounted as volume - no COPY needed!
# This means instant code changes without rebuilds

EXPOSE 8000

# Default development command with auto-reload
CMD ["python", "-m", "flask", "run", "--host=0.0.0.0", "--port=8000", "--reload", "--debugger"]
